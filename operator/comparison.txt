주요 비교 연산자

$eq
- { age: { $eq: 25 } } // age가 25인 문서 검색

$ne
- { status: { $ne: "active" } } // status가 "active"가 아닌 문서 검색

$gt
- { age: { $gt: 30 } } // age가 30 초과인 문서 검색

$gte
- { age: { $gte: 30 } } // age가 30 이상인 문서 검색

$lt
- { age: { $lt: 30 } } // age가 30 미만인 문서 검색

$lte
- { age: { $lte: 30 } } // age가 30 이하인 문서 검색

$in
- { age: { $in: [28, 29, 30] } } // age가 $in의 키값(배열)내에 존재하는 문서 검색

$nin
- { age : { $nin: [28, 29, 30] } } // age가 $nin의 키값(배열)내에 존재하지 않는 문서 검색
